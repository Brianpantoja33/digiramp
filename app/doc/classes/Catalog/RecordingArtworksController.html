def destroy
  forbidden unless current_catalog_user.update_recording
  if recording_item = RecordingItem.where(itemable_type: 'Artwork', 
                                          itemable_id: params[:id]).first
    recording_item.destroy!                                        
  end
  @remove_tag = "#artwork_#{params[:id]}"
end
