def add_found

  @recordings     = Recording.not_in_bucket.find_in_collection(@catalog, @account, session[:query] )
  
  session[:query] = nil
  
  if @recordings
    @recordings.each do |recording|
      @catalog.attach_recording recording
    end
  end
  
  
end
