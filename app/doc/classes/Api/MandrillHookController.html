def handle_clicks_and_opens event
  
  # '--- handle handle_clicks_and_opens  ----'
  if message = get_by_id( event )

    if msg = event["msg"]
      if opens =  msg["opens"]
        message.opens   = opens.count
      end
      if clicks = msg["clicks"]
        message.clicks = clicks.count
      end
      message.save

    end
  end
end
