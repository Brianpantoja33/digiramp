def edit
  @shop_order   = current_order
  if error_message = @shop_order.invalid?
    flash[:danger] = error_message
    redirect_to shop_order_path( @shop_order )
  else
    @user         = current_user
    @shop_order.reset!
    @total_price  = @shop_order.total_price
    @years        = CreditCard.years
    @months       = CreditCard.months
  end

end

def payment_status
  @shop_order = Shop::Order.cached_find(params[:uuid])
  render nothing: true, status: 404 and return unless @shop_order
  render json: {guid: @shop_order.id, status: @shop_order.state, error: @shop_order.error}
end

def update
 
  params[:shop_order][:email]           = params[:email]
  params[:shop_order][:stripe_token]    = params[:stripeToken]
  params[:stripeToken]                  = nil
  
  if @shop_order = Shop::Order.cached_find(params[:id])

    if @shop_order.update(shop_order_params)
      StripeChargeJob.perform_later(@shop_order.id)
      render json: { uuid: @shop_order.id }
    else
      
      errors = @shop_order.errors.full_messages
      message = "Shop::OrdersController#update: #{errors}"
      ErrorNotification.post message

      render json: {
                    error: errors.join(" ")
                    }, status: 400
    end
  else
    message = "Shop::OrdersController#update: Session expired, please reload page and try again"

    ErrorNotification.post message
    render json: {
                  error: 'Session expired, please reload page and try again'
                  }, status: 400
  end

end
