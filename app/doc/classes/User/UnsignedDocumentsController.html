def index
  document_ids = @user.document_users
                      .where(signed_on: nil)
                      .map {|document_user| document_user.document_id  unless document_user.document_id.nil?}
  
  document_ids.uniq!

  @documents = Document.order(:title).where(uuid: document_ids)
end
