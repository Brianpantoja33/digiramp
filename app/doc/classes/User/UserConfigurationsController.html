def update
  
  if params[:commit] == "Don't ask me again"
    @user_configuration.deactivated!
  else
    @user_configuration.activated!
    @user_configuration.update(user_configuration_params)
    @user_configuration.reset!
  end
  
  
  redirect_to @user
