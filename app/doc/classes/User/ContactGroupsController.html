def add_all
 
  
  @user.clients.each do |client|
    ClientGroupsClients.where(client_id: client.id, client_group_id: params[:contact_group_id])
                       .first_or_create(client_id: client.id, client_group_id: params[:contact_group_id])
    
  end

  render nothing: true
end
