def remove_notification_message user_id
  if notification_message = NotificationMessage.find_by(
      user_id: user_id,
      asset_id: self.id,
      asset_type: self.class.name
    )
    notification_message.destroy
  end
end

def update_notification_message user_id

    asset_id: self.id,
    asset_type: self.class.name,
  )
  .first_or_create(
